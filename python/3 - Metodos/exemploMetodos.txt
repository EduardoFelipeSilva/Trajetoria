def visualizarTabuada():
    numero = int(input("Digite o n√∫mero da tabuada: "))
    contador = 0

def conversaoCelsiusFahrenheit(celsius):
    fah = celsius * 1.8 + 32
    print(f"C {fah:.2f} ")
