class Pessoa:

        self.__nome = nome
        self.__idade = idade
        self.__peso = peso

    def cadastrarPessoa(self):
        self.__nome = input('Digite seu nome: ')
        self.__idade = int(input('Digite sua idade: '))
        self.__peso = float(input('Digite seu peso:'))

    def exibirPessoa(self):
        print(self.__nome, self.__idade, self.__peso)
